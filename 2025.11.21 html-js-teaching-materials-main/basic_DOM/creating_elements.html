Creating elements

This example creates an element and adds it to the DOM (HTML body document) when the page loads.

function handleClickEvent(event) { //runs when you click on element const replaceElement = document.querySelector('#replaceMe'); replaceElement.innerHTML = "Replaced! " + Date.now(); // get a list of all .info elements const infoTextElements = document.querySelectorAll(".info"); for (const elem of infoTextElements) { elem.classList.replace("info", "info2"); } // stop listening event.target.classList.add("deactivated"); //event target, thing that is clicked event.target.removeEventListener('click',handleClickEvent); }
